#include<bits/stdc++.h> 
#include<iostream>
using namespace std;

unsigned const int mod = 10007;

void solve(long long int n, long long int &ans){
    // cur represents the current factorial value
	long long int cur = 1;
	for(int i=1;i<=n;++i){
		cur *= i;  
        cur %= mod; // take modulo, preventing from overflow
        ans += cur;
        ans %= mod; // take modulo, preventing from overflow
	}
}

int main(){
	// setting up variables
	long long int n,ans=0;
	// take input
	cin >> n;
	// execute corresponding function
	solve(n,ans);
	// output
	cout << ans;
	return 0;
}
