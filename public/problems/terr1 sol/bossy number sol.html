#include<bits/stdc++.h> 
using namespace std;

bool solve(int a){
	// check if it's a perfect square
	if(pow((int)(pow(a,0.5)),2) == a) return true;
	else return false;
}

int main(){
	// setting up variables
	int a;
	bool b;
	// take input
	cin >> a;
	// execute corresponding function
	b = solve(a);
	// output
	cout << b;
	return 0;
}
