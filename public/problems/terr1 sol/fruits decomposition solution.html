#include<bits/stdc++.h> 
using namespace std;

// check if input is prime
bool isprime(long long int n){
	for(int i=2;i<=(int)(pow(n,0.5));++i)
		if(n % i == 0)
			return false;
	return true;
}

void solve(long long int n, long long int &ans){
	// break if n is prime
	while(!isprime(n)){
		for(int i=2;i<n;++i){
			if(n % i == 0 && isprime(i)){
				// i is n's prime factor
				ans += i;
				n = n/i;
				break;
			}
		}
	}
	// n > 1: last n is a prime factor
	ans += n > 1? n : 0;
	
}

int main(){
	// setting up variables
	long long int n,ans;
	// take input
	cin >> n;
	// execute corresponding function
	solve(n,ans);
	// output
	cout << ans;
	return 0;
}
