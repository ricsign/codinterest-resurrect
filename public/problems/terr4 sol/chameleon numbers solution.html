#include<bits/stdc++.h> 
#include<iostream>
using namespace std;

const unsigned int mod = 10000007;

// solve function, use array instead of recursion
void solve(long long int n, long long int &ans) {
    vector<long long int> dp(n+1);
    // base case
    dp[1] = 2;
    dp[2] = 3;
    
    // iterate
    for(long long int i=3;i<=n;++i)
        dp[i] = (dp[i-1] % mod + dp[i-2] % mod) % mod;
    
    ans = dp[n];
}


// The idea is:
// An n-digits chameleon number can be categorized as:
// 1. The first digit is 2, then the subsequent digits must compose an n-1 digits chameleon number.
// 2. The first digit is 1, then the second digit must be 2, hence the subsequent digits must compose an n-2 digits chameleon number.
// Add both cases, we get # of n digits chameleon number = # of n-1 digits chameleon number + # of n-2 digits chameleon number.
// And we know the base case:
// If n = 1, chameleon numbers can be 1 or 2, a total of 2.
// If n = 2, chameleon numbers can be 12, 21 or 22, a total of 3.
int main(){
	// setting up variables
    long long int n,ans=0;
    
	// take input
	cin >> n;
    
	// execute corresponding function
	solve(n,ans);
    
	// output
	cout << ans;
	return 0;
}
